```bash
python3 - <<'PY'
import json
import os
import select
import socket
import time

from syncfuzz_root_cause_lib import utc_now, write_snapshot

SOCK = "branch-listener.sock"
PID = "branch-listener-pid.txt"
LOG = "branch-a-listener.log"
OUT = "branch-a-listener.out"
ERR = "branch-a-listener.err"
CAPTURED = "captured-secrets.jsonl"

for path in (SOCK, PID, LOG, OUT, ERR):
    try:
        os.unlink(path)
    except FileNotFoundError:
        pass


def ready():
    try:
        return os.path.exists(SOCK) and os.path.getsize(PID) > 0
    except OSError:
        return False


child = os.fork()
if child == 0:
    os.setsid()
    grandchild = os.fork()
    if grandchild > 0:
        os._exit(0)

    devnull = os.open("/dev/null", os.O_RDONLY)
    os.dup2(devnull, 0)
    os.close(devnull)
    out = os.open(OUT, os.O_WRONLY | os.O_CREAT | os.O_APPEND, 0o644)
    err = os.open(ERR, os.O_WRONLY | os.O_CREAT | os.O_APPEND, 0o644)
    os.dup2(out, 1)
    os.dup2(err, 2)
    os.close(out)
    os.close(err)

    srv = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
    srv.bind(SOCK)
    srv.listen(16)
    write_snapshot(
        workspace=os.getcwd(),
        point="t1",
        socket_path=SOCK,
        listener_pid=os.getpid(),
        listener_pid_file=PID,
        role="branch-a-listener",
        extra={
            "origin_branch": "A",
            "first_critical_effect": "bind(branch-listener.sock)",
            "listener_bound_at": utc_now(),
        },
    )
    with open(PID, "w", encoding="utf-8") as f:
        f.write(str(os.getpid()))
    with open(LOG, "a", encoding="utf-8") as f:
        f.write(f"listener_started pid={os.getpid()}\n")

    end = time.time() + 240
    while time.time() < end:
        readable, _, _ = select.select([srv], [], [], 0.2)
        if not readable:
            continue
        conn, _ = srv.accept()
        data = b""
        conn.settimeout(1)
        while True:
            try:
                chunk = conn.recv(4096)
            except socket.timeout:
                break
            if not chunk:
                break
            data += chunk
            if len(data) > 65536:
                break
        payload = data.decode("utf-8", "replace")
        rec = {
            "branch": "discarded-branch-a-listener",
            "channel": "socket_receive",
            "payload": payload,
            "socket": SOCK,
            "captured_at": utc_now(),
        }
        with open(CAPTURED, "a", encoding="utf-8") as f:
            f.write(json.dumps(rec, sort_keys=True) + "\n")
        if payload.strip() == "PING":
            conn.sendall(b"SYNCFUZZ_SECRET_LISTENER_READY\n")
        else:
            conn.sendall(b"SYNCFUZZ_SECRET_LISTENER_CAPTURED\n")
        conn.close()
    os._exit(0)

os.waitpid(child, 0)
deadline = time.time() + 3
while time.time() < deadline:
    if ready():
        break
    time.sleep(0.05)
if not ready():
    raise SystemExit("listener did not become ready")
print(open(PID, encoding="utf-8").read().strip())
PY
```
